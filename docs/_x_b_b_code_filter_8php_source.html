235  /**
236  * Build the tag tree from a text.
237  *
238  * @param string $text
239  * The source text to parse.
240  *
241  * @return array
242  * Two values: The tree and a (unique) array of all tag names encountered.
243  */
244  private function buildTree($text) {
245  // Find all opening and closing tags in the text.
246  preg_match_all(XBBCODE_RE_TAG, $text, $matches, PREG_SET_ORDER | PREG_OFFSET_CAPTURE);
247 
248  // Initialize the name tracker, and the list of valid tags.
249  $open_by_name = [];
250  $tags = [];
251  $found_tags = [];
252  foreach ($matches as $match) {
253  $tag = new Element($match);
254  if ($this->tagsByName($tag->name)) {
255  $tag->selfclosing = $this->tagsByName($tag->name)->isSelfclosing();
256  $tags[] = $tag;
257  $open_by_name[$tag->name] = 0;
258  }
259  }
260 
261  // Initialize the stack with a root element.
262  $stack = [new RootElement()];
263  foreach ($tags as $tag) {
264  // Add text before the new tag to the parent.
265  end($stack)->advance($text, $tag->start);
266 
267  // Case 1: The tag is opening and not self-closing.
268  if (!$tag->closing && !$tag->selfclosing) {
269  // Stack the open tag, and increment the tracker.
270  array_push($stack, $tag);
271  $open_by_name[$tag->name]++;
272  }
273 
274  // Case 2: The tag is self-closing.
275  elseif ($tag->selfclosing) {
276  end($stack)->append($tag, $tag->end);
277  $found_tags[$tag->name] = $tag->name;
278  }
279 
280  // Case 3: The tag closes an existing tag.
281  elseif ($open_by_name[$tag->name]) {
282  $open_by_name[$tag->name]--;
283 
284  // Find the last matching opening tag, breaking everything after it.
285  while (end($stack)->name != $tag->name) {
286  $dangling = array_pop($stack);
287  end($stack)->breakTag($dangling);
288  $open_by_name[$dangling->name]--;
289  }
290  $current = array_pop($stack);
291  $current->advance($text, $tag->start);
292  $current->source = substr($text, $current->end, $current->offset - $current->end);
293  $current->closer = $tag;
294  end($stack)->append($current, $tag->end);
295  $found_tags[$tag->name] = $tag->name;
296  }
297  }
298 
299  // Add the remainder of the text, and then break any tags still open.
300  end($stack)->advance($text, strlen($text));
301  while (count($stack) > 1) {
302  $dangling = array_pop($stack);
303  end($stack)->breakTag($dangling);
304  }
305  return [end($stack), $found_tags];
306  }
