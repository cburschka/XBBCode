Detailed Description

Represents a custom XBBCode tag that can be altered by administrators.

( id = "xbbcode_tag", label = ("BBCode custom tag"), handlers = { "form" = { "add" = "Drupal\xbbcode\Form\TagAddForm", "edit" = "Drupal\xbbcode\Form\TagEditForm", "delete" = "Drupal\xbbcode\Form\TagDeleteForm" }, "list_builder" = "Drupal\xbbcode\TagListBuilder", "access" = "Drupal\xbbcode\TagAccessHandler" }, config_prefix = "tag", admin_permission = "administer custom BBCode tags", entity_keys = { "id" = "id", "label" = "label", "status" = "status" }, links = { "edit-form" = "/admin/config/content/xbbcode/tags/manage/{xbbcode_tag}", "delete-form" = "/admin/config/content/xbbcode/tags/manage/{xbbcode_tag}/delete" }, config_export = { "id", "label", "description", "sample", "name", "selfclosing", "attached", "editable", "template_code", "template_file" } )
