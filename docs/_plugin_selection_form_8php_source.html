156  public static function validateTags(array $element, FormStateInterface $form_state) {
157  // Generate the prefix path of the form element.
158  $parents = implode('][', $element['#parents']);
159 
160  $names = [];
161  $errors = [];
162 
163  foreach ($form_state->getValue($element['#parents']) as $id => $plugin) {
164  if (!preg_match('/^[a-z0-9_]+$/', $plugin['name'])) {
165  $form_state->setErrorByName("{$parents}][{$id}][name", t('The name [%name] must consist of lower-case letters, numbers and underscores.', ['%name' => $plugin['name']]));
166  }
167  if ($plugin['status']) {
168  // Track which plugins are using which names.
169  if (!empty($names[$plugin['name']])) {
170  $errors[$plugin['name']] = $plugin['name'];
171  }
172  $names[$plugin['name']][$id] = $id;
173  }
174  }
175 
176  foreach ($errors as $name) {
177  foreach ($names[$name] as $id) {
178  $form_state->setErrorByName("{$parents}][{$id}][name", t('The name [%name] is used by multiple tags.', ['%name' => $name]));
179  }
180  }
181  }
